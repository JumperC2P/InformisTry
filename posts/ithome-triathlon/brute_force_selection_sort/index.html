// Output: An array Array[0...n-1] sorted in ascending order
function SelectionSort(Array[0, ..., n-1]){
    for i = 0 to n-2 {
        min = i
        for j = i + 1 to n-1 {
            if Array[j] < Array[min]{
                min = j
            }
        }
        SWAP Array[i] and Array[min]
    }
}
