function InsertionSort(Array[0, ..., n-1]){
    for i = 1 to n-1 {
        v = Array[i]
        j = i - 1
        while j >= 0 and Array[j] > v {
            Array[j+1] = Array[j]
            j = j - 1
        }
        A[j+1] = v
    }
}
